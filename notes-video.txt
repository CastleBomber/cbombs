# Converting video to a smaller format size
ffmpeg -i sky-bg-short.mp4 -vf "fps=24,scale=-1:720" -vcodec libx264 -crf 28 -preset fast -an -movflags +faststart sky-bg-short-compressed.mp4

# Mobile version
ffmpeg -i sky-bg-short-compressed.mp4 -vf "scale=720:-2" -vcodec libx264 -crf 28 -preset fast -an sky-bg-short-compressed-mobile.mp4
