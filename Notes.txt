# From my initial git add ., i had received the error
# fatal: Need to specify how to reconcile divergent branches.
# error: failed to push some refs to 'https://github.com/CastleBomber/website-cbombs'
Solution:
git pull origin main --no-rebase


# Tree command to go 3 levels of subfolders deep
tree -d -L 3


git branch -a
* main                     # You're on the main branch (local)
  remotes/origin/HEAD -> origin/main  # Default remote branch
  remotes/origin/gh-pages  # Auto-generated by `npm run deploy` (for your live site)
  remotes/origin/main      # Remote copy of your main branch
